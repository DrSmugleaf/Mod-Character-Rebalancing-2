require "modinfo"
package.path = package.path .. ";../mods/workshop-#/scripts/prefabs/?.lua"
require "CHARACTER"

local function balanceCHARACTERstats(inst)
	local CHARACTERstats =	{
								health = 75,
								levelingfood = {monsterlasagna = 1}
							}

	ModifyCharacter:ModifyStats(inst, characterstats)
	OR
	ModifyCharacter:ModifyStatsWithLeveling(inst, characterstats)
end

if not ignoreMCR then
	AddPrefabPostInit("CHARACTER", balanceCHARACTERstats)
	LogHelper:PrintInfo("Balancing CHARACTER by " .. author .. ": Version " .. version)
else
	LogHelper:PrintInfo("Balancing CHARACTER disabled by author")
end
