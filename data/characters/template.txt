package.path = package.path .. ";../mods/workshop-#/?.lua"
require "modinfo"
package.path = package.path .. ";../mods/workshop-#/scripts/prefabs/?.lua"
require "CHARACTER"

local function balancestats(inst)
	local characterstats =	{
								health = 75,
								levelingfood = {monsterlasagna = 1}
							}

	ModifyCharacter:ModifyStats(inst, characterstats)
	OR
	ModifyCharacter:ModifyStatsWithLeveling(inst, characterstats)
end

if not ignoreMCR then
	if version ~= TUNING.CHARACTER_SUPPORTED_VERSION then
		LogHelper:PrintWarn("Running unsupported version of " .. name .. " Version: " .. version .. " Supported version: " .. TUNING.CHARACTER_SUPPORTED_VERSION)
	end
	LogHelper:PrintInfo("Balancing " .. name ..  " by " .. author .. " Version: " .. version)
	AddPrefabPostInit("CHARACTER", balancestats)
else
	LogHelper:PrintInfo("Balancing " .. name .. " disabled by " .. author)
end
